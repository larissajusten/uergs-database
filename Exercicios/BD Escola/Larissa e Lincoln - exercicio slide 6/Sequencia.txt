#Listar todos os alunos que são do curso ENG. COMPUTAÇÃO

#Listar todos os alunos que são do curso SISTEMAS EMBARCADOS
select aluno.nome from aluno where aluno.nroMatricula in (SELECT matricula.nroMatricula FROM matricula WHERE matricula.idTurma in (select turma.idTurma from turma where turma.idCurso in (SELECT idCurso from curso where descricao LIKE '%Embarcados')))

#Listar todos os alunos que são do curso PEDAGOGIA
select aluno.nome from aluno where aluno.nroMatricula in (SELECT matricula.nroMatricula FROM matricula WHERE matricula.idTurma in (select turma.idTurma from turma where turma.idCurso in (SELECT idCurso from curso where descricao LIKE 'Pedagogia')))

#Listar todos os alunos que são do curso AUTOMAÇÃO
select aluno.nome from aluno where aluno.nroMatricula in (SELECT matricula.nroMatricula FROM matricula WHERE matricula.idTurma in (select turma.idTurma from turma where turma.idCurso in (SELECT idCurso from curso where descricao LIKE 'automa%')))

#(EXTRA)Listar todas as turmas que iniciaram em MARÇO e terminaram em JULHO
